Syntax

oP = orientationPlot(ax,CS1,varargin)

Example

cs = crystalSymmetry('mmm')
oS = axisAngleSections(cs,cs)
ori = oS.makeGrid('resolution',2*degree);
oM = PatalaColorKey(cs,cs)
rgb = oM.orientation2color(ori);
plot(oS,rgb,'surf')
